ListNode? addTwoNumbers(ListNode? l1, ListNode? l2) {
  ListNode? tempL1 = l1;
  ListNode? tempL2 = l2;
  ListNode? head;
  ListNode? tail;
  int sum;
  int carry = 0;
  while (tempL1 != null || tempL2 != null){
    sum = (tempL1?.val ?? 0) + (tempL2?.val ?? 0) + carry;
    carry = sum ~/ 10;
    sum = sum % 10;

    if (tail == null){
      head = ListNode(sum);
      tail = head;
    } else {
      tail.next = ListNode(sum);
      tail = tail.next;
    }
    tempL1 = tempL1?.next;
    tempL2 = tempL2?.next;
  }
  if (carry > 0){
    tail?.next = ListNode(carry);
  }
  return head;
}
